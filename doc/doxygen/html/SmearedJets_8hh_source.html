120  CmpState compare(const Projection& p) const {
121  // Compare truth jets definitions
122  const CmpState teq = mkPCmp(p, "TruthJets");
123  if (teq != CmpState::EQ) return teq;
124 
125  // Compare lists of detector functions
126  const SmearedJets& other = dynamic_cast<const SmearedJets&>(p);
127  const CmpState nfeq = cmp(_detFns.size(), other._detFns.size());
128  if (nfeq != CmpState::EQ) return nfeq;
129  for (size_t i = 0; i < _detFns.size(); ++i) {
130  const CmpState feq = _detFns[i].cmp(other._detFns[i]);
131  if (feq != CmpState::EQ) return feq;
132  }
133 
134  // If we got this far, we're equal
135  return CmpState::EQ;
136  }

140  void project(const Event& e) {
141  // Copying and filtering
142  const Jets& truthjets = apply<JetFinder>(e, "TruthJets").jetsByPt(); //truthJets();
143  _recojets.clear(); _recojets.reserve(truthjets.size());
144  // Apply jet smearing and efficiency transforms
145  for (const Jet& j : truthjets) {
146  Jet jdet = j;
147  bool keep = true;
148  MSG_DEBUG("Truth jet: " << "mom=" << jdet.mom()/GeV << " GeV, pT=" << jdet.pT()/GeV << ", eta=" << jdet.eta());
149  for (const JetEffSmearFn& fn : _detFns) {
150  double jeff = -1;
151  std::tie(jdet, jeff) = fn(jdet); // smear & eff
152  // Re-add constituents & tags if (we assume accidentally) they were lost by the smearing function
153  if (jdet.particles().empty() && !j.particles().empty()) jdet.particles() = j.particles();
154  if (jdet.tags().empty() && !j.tags().empty()) jdet.tags() = j.tags();
155  MSG_DEBUG(" ->" << "mom=" << jdet.mom()/GeV << " GeV, pT=" << jdet.pT()/GeV << ", eta=" << jdet.eta());
156  // MSG_DEBUG("New det jet: "
157  // << "mom=" << jdet.mom()/GeV << " GeV, pT=" << jdet.pT()/GeV << ", eta=" << jdet.eta()
158  // << ", b-tag=" << boolalpha << jdet.bTagged()
159  // << ", c-tag=" << boolalpha << jdet.cTagged()
160  // << " : eff=" << 100*jeff << "%");
161  if (jeff <= 0) { keep = false; break; } //< no need to roll expensive dice (and we deal with -ve probabilities, just in case)
162  if (jeff < 1 && rand01() > jeff) { keep = false; break; } //< roll dice (and deal with >1 probabilities, just in case)
163  }
164  if (keep) _recojets.push_back(jdet);
165  }
166  // Apply tagging efficiencies, using smeared kinematics as input to the tag eff functions
167  for (Jet& j : _recojets) {
168  // Decide whether or not there should be a b-tag on this jet
169  const double beff = _bTagEffFn ? _bTagEffFn(j) : j.bTagged();
170  const bool btag = beff == 1 || (beff != 0 && rand01() < beff);
171  // Remove b-tags if needed, and add a dummy one if needed
172  if (!btag && j.bTagged()) j.tags().erase(std::remove_if(j.tags().begin(), j.tags().end(), hasBottom), j.tags().end());
173  if (btag && !j.bTagged()) j.tags().push_back(Particle(PID::BQUARK, j.mom()));
174  // Decide whether or not there should be a c-tag on this jet
175  const double ceff = _cTagEffFn ? _cTagEffFn(j) : j.cTagged();
176  const bool ctag = ceff == 1 || (ceff != 0 && rand01() < beff);
177  // Remove c-tags if needed, and add a dummy one if needed
178  if (!ctag && j.cTagged()) j.tags().erase(std::remove_if(j.tags().begin(), j.tags().end(), hasCharm), j.tags().end());
179  if (ctag && !j.cTagged()) j.tags().push_back(Particle(PID::CQUARK, j.mom()));
180  }
181  }
