114  Vector3 closestApproach() const {
115  const FourVector& v0 = origin();
117  const double rho0 = origin().perp() / sin(this->phi() - origin().phi());
118  const double phi0 = M_PI/2 - this->phi();
119  const double x0 = rho0 * cos(phi0);
120  const double y0 = rho0 * sin(phi0);
121  const double z0 = origin().z() - v0.perp()/tan(this->theta());
122  return Vector3(x0, y0, z0);
